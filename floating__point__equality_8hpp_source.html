38 // https://stackoverflow.com/questions/19837576/comparing-floating-point-number-to-zero
39 // see Knuth section 4.2.2 pages 217-218
40 template<typename T>
41 bool is_almost_equal(T x, T y) {
42  return std::abs(x - y) <= std::numeric_limits < T > ::epsilon() * std::abs(x);
43 }

45 /**
46  * TODO: Passing a fixed limit like below is bad practice...

48  * std::numeric_limits<typename ImageT::value_type>::epsilon() * std::abs(img1(x,y));
49  */
50 static bool is_almost_equal(const Image &img1, const Image &img2,
51  double acceptable_tolerance = 0.00001) {
52  bool same_width = (img1.width() == img2.width());
53  bool same_height = (img1.height() == img2.height());
54  bool same_size = same_width && same_height;
55 
56  if (!same_size) {
57  return false;
58  }
59 
60  Image sub = img1 - img2;
61 
62  cimg_forXY(sub, x, y)
63  {
64  bool considerEqual = (std::abs(sub(x, y)) <= acceptable_tolerance);
65 
66  if (!considerEqual) {
67  return false;
68  }
69  }
70 
71  return true;
72 }
