122  /**
123  * Calculate histogram idx from pixel value.
124  *
125  * Example:
126  *
127  * num_bins = 4
128  * lower_boundary = 0, upper_boundary = 99 -> delta_max_min = 100
129  *
130  * rel_pos = value / delta_max_min
131  * idx = num_bins * rel_pos
132  *
133  * value = 0 -> rel_pos = 0 / 100 = 0.0 -> idx = 4 * 0.00 = 0.00 -> 0
134  * value = 10 -> rel_pos = 10 / 100 = 0.1 -> idx = 4 * 0.10 = 0.40 -> 0
135  * value = 25 -> rel_pos = 25 / 100 = 0.25 -> idx = 4 * 0.25 = 1.00 -> 1
136  * value = 30 -> rel_pos = 30 / 100 = 0.3 -> idx = 4 * 0.30 = 1.20 -> 1
137  * value = 60 -> rel_pos = 60 / 100 = 0.6 -> idx = 4 * 0.60 = 2.40 -> 2
138  * value = 99 -> rel_pos = 99 / 100 = 0.99 -> idx = 4 * 0.99 = 3.96 -> 3
139  */
140  size_t calculate_histogram_idx_from_pixel_value_internal(
141  ImageType pixel_value) const {
142 
143  throw_if_lower_boundary_is_less_or_equal_upper_boundary_value();
144  throw_if_lower_boundary_is_greater_than_min_image_pixel();
145  throw_if_upper_boundary_is_less_than_max_image_pixel();
146 
147  const ImageType delta_max_min = upper_boundary_ - lower_boundary_ + 1;
148  ImageType pixel_value_minus_min = pixel_value - lower_boundary_;
149  float factor = (float) pixel_value_minus_min / (float) delta_max_min;
150  size_t num_bins = histogram_.size();
151 
152  return (size_t) ((float) num_bins * factor);
153  }
154 
155  /**
156  *
157  */
158  ImageType calculate_pixel_value_from_histogram_idx_internal(
159  size_t idx) const {
160  throw_if_idx_exceeds_num_bins(idx);
161  throw_if_lower_boundary_is_less_or_equal_upper_boundary_value();
162  throw_if_lower_boundary_is_greater_than_min_image_pixel();
163  throw_if_upper_boundary_is_less_than_max_image_pixel();
164 
165  const ImageType delta_max_min = upper_boundary_ - lower_boundary_ + 1;
166  size_t num_bins = histogram_.size();
167  float factor = (float) idx / (float) num_bins;
168  float pixel_value_minus_min = factor * (float) delta_max_min;
169 
170  return pixel_value_minus_min + lower_boundary_;
171  }

177  const cimg_library::CImg<ImageType> &input_image, size_t num_bins) {
178 
179  throw_if_num_bins_not_valid(num_bins);

181  histogram_.resize(num_bins, 0);

183  // TODO: Check that input_image(x, y) - min_pixel_value does not get smaller than image.min()

186  /**
187  * IDEA: "Shrink" /map the float image pixel values to 256 possible
188  * brightness levels (i.e. 256 histogram bins). The reason is that
189  * the performance of the "max entropy algorithm" strongly depends
190  * on the histogram size / number of bins. Note that the threshold
191  * which will be calculated based on this histogram will be the
192  * correct one for this "shrunk" histogram. In order to get the
193  * threshold for the initial float image, this transformation needs
194  * to be reverted later (see comment below).
195  */

197  cimg_forXY(input_image, x, y)
198  {

200  input_image(x, y));

202  ++histogram_[idx];
203  }
