55 /**
56  * Define a cost functor for Gaussian curve fitting.
57  * NOTE: Supplying vectors of data is more flexible
58  * - e.g. x, y and z data... Therefore, Point is not
59  * used here.
60  */
61 struct GaussianResidual {
62  GaussianResidual(double x, double y)
63  :
64  x_(x),
65  y_(y) {
66  }
67  template<typename T>
68  bool operator()(const T *const params, T *residual) const {
69  T A = params[0];
70  T mu = params[1];
71  T sigma = params[2];
72  residual[0] = y_ - A * exp(-0.5 * pow((x_ - mu) / sigma, 2));
73  return true;
74  }
75  private:
76  const double x_;
77  const double y_;
78 };

111 /**
112  * Calc FWHM ["] from FWHM [px]
113  * FWHM = sigma * 1.66510922 (=2*sqrt(ln(2)))
114  *
115  * See http://mathworld.wolfram.com/GaussianFunction
116  *
117  * NOTE: std::log() is ln()
118  * TODO: We want to have this value in PX and ".... depending on the purpose...
119  */
120 static inline double sigma_to_fwhm(double sigma) {
121  constexpr double SIGMA_TO_FWHM = 2.0 * std::sqrt(std::log(2.0));
122  return SIGMA_TO_FWHM * sigma;
123 }

128 template<typename ImageType>
129 std::vector<double> make_guess(const std::vector<ImageType> &input_values) {
130  // TODO: THROW_IF(LmCurveMatcher, input_values.size() < 2, "input_values.size() < 2!")
131 
132  auto max_element_iter = std::max_element(input_values.begin(),
133  input_values.end());
134  return std::vector<double> { *max_element_iter, /* max. y value -> A*/
135  (double) std::distance(input_values.begin(), max_element_iter), /*x value of max. y value -> mu */
136  input_values.size() / 10.0 /* "x-range" of values divided by 10 */
137  };
138 }

162  for (size_t i = 0; i < x_data.size(); ++i) {
163  problem.AddResidualBlock(
