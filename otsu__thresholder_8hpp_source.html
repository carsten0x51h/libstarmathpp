38 /**
39  * Otsu’s method assumes that our image contains two classes of pixels: the background and
40  * the foreground. Furthermore, Otsu’s method makes the assumption that the grayscale
41  * histogram of our pixel intensities of our image is bi-modal, which simply means that
42  * the histogram is two peaks.
43  *
44  * Otsu's thresholding method involves iterating through all the possible threshold
45  * values and calculating a measure of spread for the pixel levels each side of the
46  * threshold, i.e. the pixels that either fall in foreground or background. The aim
47  * is to find the threshold value where the sum of foreground and background spreads
48  * is at its minimum.
49  *
50  * It’s also important to note that Otsu’s method is an example of global thresholding
51  * — implying that a single thresholding value is computed for the entire image.
52  *
53  * This threshold is determined by minimizing intra-class intensity variance, or
54  * equivalently, by maximizing inter-class variance. The algorithm exhaustively
55  * searches for the threshold that minimizes the intra-class variance, defined
56  * as a weighted sum of variances of the two classes
57  *
58  * See https://rndayala.wordpress.com/2019/11/13/image-processing-thresholding/
59  * See http://www.labbookpages.co.uk/software/imgProc/otsuThreshold.html
60  *
61  * IDEA / TODO: int BitDepth as second template parameter?
62  * TODO: Use Histogram class?
63  */

88  [[nodiscard]] float calculate_threshold(
89  const cimg_library::CImg<ImageType> &input_image) const override {
90 
91  if (input_image.width() <= 0 || input_image.height() <= 0) {
92  throw ThresholderException("No image supplied.");
93  }
94 
95  auto num_buckets = (size_t) std::pow(2.0F, bit_depth_);
96  std::vector < ImageType > hist(num_buckets, 0.0F);
97 
98  float num_pixels = input_image.width() * input_image.height();
99  float sum_b = 0.0F;
100  float wb = 0.0F;
101  float max = 0.0F;
102  float threshold1 = 0.0F;
103  float threshold2 = 0.0F;
104 
105  // Calculate histogram - for some reason inImg.get_histogram() behaves unexpectedly.
106  cimg_forXY(input_image, x, y)
107  {
108  int value = (int) input_image(x, y);
109  ++hist[value];
110  }
111 
112  float sum = 0;
113  for (size_t pos = 0; pos < num_buckets; ++pos) {
114  sum += (float) pos * hist[pos];
115  }
116 
117  for (size_t i = 0; i < num_buckets; ++i) {
118  wb += hist[i];
119 
120  if (is_almost_equal(wb, 0.0F)) {
121  continue;
122  }
123 
124  float wf = num_pixels - wb;
125 
126  if (is_almost_equal(wf, 0.0F)) {
127  break;
128  }
129 
130  sum_b += (float) i * hist[i];
131 
132  float mF = (sum - sum_b) / wf;
133  float mB = sum_b / wb;
134  float diff = mB - mF;
135  float bw = wb * wf * std::pow(diff, 2.0F);
136 
137  if (bw >= max) {
138  threshold1 = (float) i;
139  if (bw > max) {
140  threshold2 = (float) i;
141  }
142  max = bw;
143  }
144  } // end loop
145 
146  return (threshold1 + threshold2) / 2.0F;
147  }
148 };
