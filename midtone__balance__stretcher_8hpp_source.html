64  /**
65  * Defined for 0 <= x <= 1
66  *
67  * See https://pixinsight.com/doc/tools/HistogramTransformation/HistogramTransformation.html
68  * See https://siril.readthedocs.io/en/latest/processing/stretching.html#midtone-transfer-function-transformation-mtf
69  * See https://pixinsight.com/doc/tools/AutoHistogram/AutoHistogram.html
70  */
71  float midtone_transfer_function(float x, float m, float lo, float hi) const {
72  if (x <= lo) {
73  return 0.0F;
74  }
75 
76  if (x >= hi) {
77  return 1.0F;
78  }
79 
80  float xp = (x - lo) / (hi - lo);
81 
82  return ((m - 1.0F) * xp) / (((2.0F * m - 1.0F) * xp) - m);
83  }

96  const cimg_library::CImg<float> &input_image, float target_background =
97  0.25F) const {

99 #define shadowsClipping -2.80f /* Shadows clipping point measured in sigma units from the main histogram peak. */
100  //#define targetBackground 0.25f /* final "luminance" of the image for autostretch in the [0,1] range */
101 
102  using namespace cimg_library;
103 
104  // The MAD is defined as the median of the absolute deviations from the
105  // data's median: MAD = median (| Xi âˆ’ median(X) |)
106  auto median = input_image.median();
107  auto median_diff_image = input_image - median;
108  auto mad = median_diff_image.abs().median();
109 
110  // this is a guard to avoid breakdown point
111  if (mad == 0.0F) {
112  mad = 0.001F;
113  }
114 
115  float shadows = clip_to_zero(median + shadowsClipping * mad);
116  float m = midtone_transfer_function(median - shadows, target_background,
117  0.0F, 1.0F);
118 
119  return {m, shadows, 1.0F};
120  }
