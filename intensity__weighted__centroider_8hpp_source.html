38 /**
39  * Intensity Weighted Centroiding (IWC) is similar to WCoG with a
40  * difference that the weighting function, Wij is the intensity
41  * distribution of the spot pattern, Iij .
42  *
43  * In comparison to the CoG method, this algorithm performs a better job
44  * under low light level conditions and low background and readout noise.
45  *
46  * The "Intensity Weighted Centroiding" is described on page 170 of
47  * "Topics in Adaptive Optics" - "Advanced Methods for Improving the Efficiency
48  * of a Shack Hartmann Wavefront Sensor"
49  *
50  * See http://cdn.intechopen.com/pdfs-wm/26716.pdf
51  */
52 template<typename ImageType>
53 class IntensityWeightedCentroider : public Centroider<ImageType> {
54  public:
55  [[nodiscard]] std::string get_name() const override {
56  return "IntensityWeightedCentroider";
57  }
58 
59  /**
60  * TODO: auto as return value?
61  */
62  [[nodiscard]] std::optional<Point<float>> calculate_centroid(
63  const cimg_library::CImg<ImageType> &input_image) const override {
64 
65  if (input_image.width() <= 0 || input_image.height() <= 0) {
66  throw CentroiderException("No image supplied.");
67  }
68 
69  Point<float> centroid;
70 
71  double I2;
72  double I2x = 0;
73  double I2y = 0;
74  double Ixy2 = 0;
75 
76  // NOTE: indices x and y fun from 0..M-1.
77  // However, for the calculation 1..M is expected.
78  // Therefore, +1 is needed. Otherwise, the first I2x
79  // and I2y are not counted.
80  cimg_forXY(input_image, x, y)
81  {
82  I2 = std::pow(input_image(x, y), 2.0);
83  I2x += I2 * (x + 1);
84  I2y += I2 * (y + 1);
85  Ixy2 += I2;
86  }
87 
88  // NOTE: Since the index of an image runs from 0..M-1,
89  // 1 needs to be subtracted again.
90  centroid.set_x((float) (I2x / Ixy2) - 1);
91  centroid.set_y((float) (I2y / Ixy2) - 1);
92 
93  return std::optional<Point<float> > { centroid };
94  }
95 };
