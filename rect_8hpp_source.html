295  static Rect<S> change_rect_size_internal(Rect<T> rect, S change_by,
296  bool grow) {
297  auto center = calc_center_from_rect_internal<float>(rect);
298 
299  S border_both_sides = 2.0 * change_by;
300 
301  S new_width = (
302  grow ?
303  (S) (rect.width() + border_both_sides) :
304  (S) clip_to_zero((S) rect.width() - border_both_sides));
305 
306  S new_height = (
307  grow ?
308  (S) (rect.height() + border_both_sides) :
309  (S) clip_to_zero((S) rect.height() - border_both_sides));
310 
311  return from_center_point_internal<float, S, S>(center, new_width,
312  new_height);
313  }

335  // Is rect1 completely contained inside this rect2?
336  // Of course both coordinates need to be in the same coordinate system.
337  static bool rect1_contains_rect2_internal(const Rect<T> &rect1,
338  const Rect<T> &rect2) {
339 
340  if (!rect1.is_set_ || !rect2.is_set_) {
341  return false;
342  }
343 
344  bool contains = rect2.x() >= rect1.x() && rect2.y() >= rect1.y()
345  && rect2.x() + rect2.width() <= rect1.x() + rect1.width()
346  && rect2.y() + rect2.height() <= rect1.y() + rect1.height();
347 
348  return contains;
