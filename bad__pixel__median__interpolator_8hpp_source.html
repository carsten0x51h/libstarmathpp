199  template<typename ImageType>
200  ImageType interpolate_internal(
201  ImageType pixel_value, const cimg_library::CImg<ImageType> &neighbourhood,
202  float threshold, ThresholdDirection::TypeE thresholdDirection) {
203  ImageType med = neighbourhood.median();
204 
205  bool want_interpolation = false;
206 
207  switch (thresholdDirection) {
208  case ThresholdDirection::POSITIVE:
209  want_interpolation = ((pixel_value - med) >= threshold);
210  break;
211 
212  case ThresholdDirection::NEGATIVE:
213  want_interpolation = ((med - pixel_value) >= threshold);
214  break;
215 
216  case ThresholdDirection::BOTH:
217  want_interpolation = (std::abs(pixel_value - med) >= threshold);
218  break;
219 
220  default: {
221  throw_unsupported_threshold_direction(thresholdDirection);
222  }
223  }
224  return (want_interpolation ? med : pixel_value);
225  }
