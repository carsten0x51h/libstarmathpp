63 /**
64  * InterpolationTypeT specifies the method which should be used to
65  * interpolate during the scaling process.
66  *
67  * https://cimg.eu/reference/structcimg__library_1_1CImg.html
68  * resize ( const int size_x,
69  * const int size_y = -100,
70  * const int size_z = -100,
71  * const int size_c = -100,
72  * const int interpolation_type = 1
73  * ...
74  *
75  * interpolation_type Method of interpolation:
76  *
77  * -1 = no interpolation: raw memory resizing.
78  * 0 = no interpolation: additional space is filled according to boundary_conditions.
79  * 1 = nearest-neighbor interpolation.
80  * 2 = moving average interpolation.

156 template<typename ImageType = float>
157 auto scale(ScaleDirection::TypeE scale_type, float scale_factor,
158  InterpolationType::TypeE interpolation_type) {
159  return ranges::views::transform(
160  [=](const std::shared_ptr<cimg_library::CImg<ImageType> > &image) {
161 
162  const cimg_library::CImg<ImageType> &input_image_ref = *image;
163 
164  DEBUG_IMAGE_DISPLAY(input_image_ref, "scale_in",
165  STARMATHPP_PIPELINE_SCALE_DEBUG);
166 
167  auto scaled_image = std::make_shared < cimg_library::CImg<ImageType>
168  > (input_image_ref);
169 
170  float factor = (
171  scale_type == ScaleDirection::UP ?
172  scale_factor : 1.0F / scale_factor);
173 
174  scaled_image->resize((int) (factor * (float) input_image_ref.width()),
175  (int) (factor * (float) input_image_ref.height()),
176  -100, -100,
177  InterpolationType::asInt(interpolation_type));
178 
179  DEBUG_IMAGE_DISPLAY(input_image_ref, "scale_out",
180  STARMATHPP_PIPELINE_SCALE_DEBUG);
181 
182  return scaled_image;
183  }
184  );
185 }
