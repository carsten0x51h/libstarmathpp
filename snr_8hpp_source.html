39 /**
40  * There are some different ways to calculate the "SNR".
41  * The following document explains in detail of how the SNR can be
42  * calculated IF an the noise can be measured somewhere in the image
43  * - e.g. in the background or at least is known:
44  *
45  * https://www2.ph.ed.ac.uk/~wjh/teaching/dia/documents/noise.pdf
46  *
47  * The CImg library provides a variance estimator function (variance_noise()).
48  * With this function a more or less typical noise is simulated
49  * so that a more complicated background extraction and noise measurement
50  * does not need to be implemented here. For the purpose of recognizing if
51  * a star is within the "selection window" or not it is sufficient.
52  *
53  * Also see
54  * - https://en.wikipedia.org/wiki/Signal-to-noise_ratio_(imaging)
55  * - https://www.researchgate.net/post/How_is_SNR_calculated_in_images
56  * - https://www.dspguide.com/ch25/3.htm
57  * - https://www2.ph.ed.ac.uk/~wjh/teaching/dia/documents/noise.pdf
58  * - https://nptel.ac.in/courses/117104069/chapter_1/1_11.html
59  * - https://nptel.ac.in/courses/117104069/chapter_1/1_12.html
60  * - https://kogs-www.informatik.uni-hamburg.de/~neumann/BV-WS-2010/Folien/BV-4-10.pdf
61  *
62  *
63  * TODO: We may extend this class so that it can also be instantiated to store the diffeent
64  * values which were required on the way to the actual SNR calculation as well as the
65  * corresponding image.
66  */
67 template<typename ImageType>
68 double snr(const cimg_library::CImg<ImageType> &input_image) {
69  // 0 = Calc. variance as "second moment" according to
70  // https://kogs-www.informatik.uni-hamburg.de/~neumann/BV-WS-2010/Folien/BV-4-10.pdf
71  double variance_of_image = input_image.variance(0);
72  double estimated_variance_of_noise = input_image.variance_noise(0);
73  double q = variance_of_image / estimated_variance_of_noise;
74  double q_clip = (q > 1 ? q : 1);
75 
76  // The simulated variance of the noise will be different from the noise in the real image.
77  // Therefore it can happen that q becomes < 1. If that happens it should be limited to 1.
78  return std::sqrt(q_clip - 1);
79 }
