126 auto crop_from_center(const Size<int> &crop_region) {
127  return ranges::view::transform(
128  [=](const std::shared_ptr<cimg_library::CImg<ImageType> > &image) {
129 
130  const cimg_library::CImg<ImageType> &input_image_ref = *image;
131 
132  DEBUG_IMAGE_DISPLAY(input_image_ref, "crop_from_center_in",
133  STARMATHPP_PIPELINE_CROP_DEBUG);
134 
135  // TODO: Maybe this calculation can be simplified...
136  Point<float> image_center((float) input_image_ref.width() / 2.0F,
137  (float) input_image_ref.height() / 2.0F);
138 
139  Rect<float> rect = Rect<float>::from_center_point(
140  image_center, crop_region.to<float>());
141 
142  // get_crop() by default applies a dirichlet boundary_condition (=0). There are other
143  // options as well. In this case, the desired behaviour is to assume that all pixel values
144  // where the defined sub-frame exceeds the image boundary are assumed to be 0.
145  //
146  // boundary_conditions = Can be { 0=dirichlet | 1=neumann | 2=periodic | 3=mirror }
147  //
148  // - Dirichlet means "0 outside image".
149  // - Neumann means "Nearest neighbor outside image" (i.e. null derivative)
150  // - Periodic means "Periodic"
151  // - Mirror means "Mirrored image outside".
152  //
153  // See https://github.com/GreycLab/CImg/issues/110
154  auto cropped_img = std::make_shared<cimg_library::CImg<ImageType>>(
155  input_image_ref.get_crop(rect.x() /*x0*/, rect.y() /*y0*/,
156  rect.x() + rect.width() - 1/*x1*/,
157  rect.y() + rect.height() - 1/*y1*/));
158 
159  DEBUG_IMAGE_DISPLAY(*cropped_img, "crop_from_center_out",
160  STARMATHPP_PIPELINE_CROP_DEBUG);
161 
162  return cropped_img;
163  }
164  );
165 }
