55 /**
56  * HDF calculation
57  * https://www.lost-infinity.com/night-sky-image-processing-part-6-measuring-the-half-flux-diameter-hfd-of-a-star-a-simple-c-implementation/
58  * https://www.lost-infinity.com/the-half-flux-diameter-hfd-for-a-perfectly-normal-distributed-star/
59  * https://www.lost-infinity.com/the-half-flux-diameter-hfd-of-a-plain-image/
60  * http://www005.upp.so-net.ne.jp/k_miyash/occ02/halffluxdiameter/halffluxdiameter_en.html
61  * http://www.cyanogen.com/help/maximdl/Half-Flux.htm
62  *
63  * Get all pixels inside a radius
64  * http://stackoverflow.com/questions/14487322/get-all-pixel-array-inside-circle
65  *
66  * TODO: Change further unit tests to DATA-TESTS..., rename / move image files
67  * TODO: Rename all .tif to .tiff
68  *
69  * TODO: Change HFD so that:
70  * -Only quadratic images are accepted
71  * -Always set outerRadius to given image dimensions -> no subimage() call anymore
72  * -Assume that the centroid is in the center of the image
73  *
74  * FwhmT fwhm = starImg.fwhm(...)
75  * HfdT hfd = starImg.hfd(...)
76  *
77  * TODO: The thresholding function should be removed completely. This processing step should be executed
78  * previously, if desired.
79  *
80  * TODO: Maybe this function should be removed, completely... It is a bit strange to have this kind of
81  * behaviour modelled in here...
82  *
83  * NOTE: This function assumes that the star center is in the center of the image.
84  * NOTE: For even image width / height an inaccuracy is introduced here because there
85  * is no real center pixel.
86  *
87  * @param inImage
88  * @param inOuterDiameter
89  * @param inScaleFactor
90  * @param outCenteredImg
91  * @param inSubBgLevel
92  * @return
93  */
94 template<typename ImageType>
95 double hfd_internal(const cimg_library::CImg<ImageType> &input_image,
96  const Point<float> &star_center,
97  unsigned int outer_hfd_diameter_px, float scale_factor) {
98 
99  if (input_image.is_empty()) {
100  throw HfdException("Empty image supplied.");
101  }
102 
103  // Crop the part from the supplied image which is needed to calculate the HFD value.
104  // A boundary check is performed.
105  Rect<int> image_bounds(0, 0, input_image.width(), input_image.height());
106  Size<unsigned int> sub_image_size(outer_hfd_diameter_px,
107  outer_hfd_diameter_px);
108 
109  // NOTE: Needs to be int because sub_image_rect x and/or y may be negative
110  // The rounding to discrete pixel coordinates takes place here.
111  Rect<int> sub_image_rect = Rect<int>::from_center_point(star_center,
112  sub_image_size);
113 
114  bool sub_image_rect_inside_bounds = image_bounds.contains(sub_image_rect);
115 
116  if (!sub_image_rect_inside_bounds) {
117  std::stringstream ss;
118  ss << "Cannot calculate HFD. Rect '" << sub_image_rect
119  << "' defined by given star center '" << star_center
120  << "' is outside image bounds '" << image_bounds << "'." << std::endl;
121 
122  throw HfdException(ss.str());
123  }
124 
125  // Finally, make a copy of the part of the image which is used to calculate the HFD.
126  Image sub_img = input_image.get_crop(
127  (int) sub_image_rect.x() /*x0*/, (int) sub_image_rect.y() /*y0*/,
128  (int) (sub_image_rect.x() + sub_image_rect.width() - 1) /*x1*/,
129  (int) (sub_image_rect.y() + sub_image_rect.height() - 1) /*y1*/);
130 
131  /**
132  * interpolation_type - see https://cimg.eu/reference/structcimg__library_1_1CImg.html
133  *
134  * -1 = no interpolation: raw memory resizing.
135  * 0 = no interpolation: additional space is filled according to boundary_conditions.
136  * 1 = nearest-neighbor interpolation.
137  * 2 = moving average interpolation.
138  * 3 = linear interpolation.
139  * 4 = grid interpolation.
140  * 5 = cubic interpolation.
141  * 6 = lanczos interpolation.
142  */
143  sub_img.resize((int) (scale_factor * (double) sub_img.width()),
144  (int) (scale_factor * (double) sub_img.height()),
145  -100 /*size_z*/, -100 /*size_c*/, 1 /*interpolation_type*/);
146 
147  // Sum up all pixel values in whole circle
148  Point<double> scaled_star_center_sub_image_px(
149  (double) sub_img.width() / 2.0, (double) sub_img.height() / 2.0);
150 
151  double scaled_outer_radius = scale_factor * outer_hfd_diameter_px / 2.0;
152  double sum_pixel_values = 0;
153  double sum_weighted_dist = 0;
154 
155  cimg_forXY(sub_img, x, y)
156  {
157  double pixel_distance_to_center = std::hypot(
158  x + 0.5 - scaled_star_center_sub_image_px.x(),
159  y + 0.5 - scaled_star_center_sub_image_px.y());
160  bool is_inside_circle = (pixel_distance_to_center <= scaled_outer_radius);
161 
162  if (is_inside_circle) {
163  sum_pixel_values += sub_img(x, y);
164  sum_weighted_dist += sub_img(x, y) * pixel_distance_to_center;
165  }
166  }
167 
168  // NOTE: Multiplying with 2 is required since actually just the HFR is calculated above
169  // NOTE: One exception is the case when there is no flux at all (i.e. a totally black image).
170  // In that case the HFD actually does not exist since there would be a division by 0.
171  // Therefore, in that situation NaN is returned.
172  return (
173  sum_pixel_values > 0.0 ?
174  (2.0 * sum_weighted_dist / sum_pixel_values) / scale_factor : NAN);
175 }
176 } // namespace detail
