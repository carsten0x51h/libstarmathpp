56  [[nodiscard]] float calculate_threshold(
57  const cimg_library::CImg<ImageType> &input_image) const override {
58 
59  if (input_image.width() <= 0 || input_image.height() <= 0) {
60  throw ThresholderException("No image supplied.");
61  }
62 
63  const size_t NUM_BINS = 256; // TODO: Do not hardcode ...
64  size_t num_bins = NUM_BINS;
65  std::vector<float> hist(num_bins, 0);
66 
67  float max;
68  float min = input_image.min_max(max);
69 
70  /**
71  * IMPORTANT / IDEA: "Shrink" /map the float image pixel values to 256 possible brightness levels (i.e. 256 histogram bins).
72  * The reason is that the performance of the "max entropy algorithm" strongly depends on the histogram size / number of bins.
73  * Note that the threshold which will be calculated based on this histogram will be the correct one for this "shrinked"
74  * histogram. In order to get the threshold for the initial float image, this transformation needs to be reverted
75  * later (see comment below).
76  */
77  cimg_forXY(input_image, x, y)
78  {
79  int idx = (int) ((float) (num_bins - 1) * (input_image(x, y) - min)
80  / (max - min + 1));
81  ++hist[idx];
82  }
83 
84  // Normalize histogram (sum of all is 1)
85  float sum = (float) input_image.width() * (float) input_image.height();
86 
87  std::vector<float> norm_hist(hist);
88  for (auto it = norm_hist.begin(); it != norm_hist.end(); ++it) {
89  *it = *it / sum;
90  }
91 
92  // Calculate accumulated histograms
93  std::vector<float> accumulated_hist_black(num_bins, 0);
94  std::vector<float> accumulated_hist_white(num_bins, 0);
95 
96  float accum_hist_sum = 0.0F;
97  for (size_t idx = 0; idx < num_bins; ++idx) {
98  accum_hist_sum += norm_hist[idx];
99 
100  accumulated_hist_black[idx] = accum_hist_sum;
101  accumulated_hist_white[idx] = 1.0F - accum_hist_sum;
102  }
103 
104  // Find first index of element not 0 in black distribution
105  size_t first_bin_idx = 0;
106 
107  for (size_t idx = 0; idx < num_bins; ++idx) {
108  if (!is_almost_equal(accumulated_hist_black[idx], 0.0F)) {
109  first_bin_idx = idx;
110  break;
111  }
112  }
113 
114  // Find last index of element not 0 in white distribution
115  size_t last_bin_idx = num_bins;
116 
117  for (size_t idx = num_bins - 1; idx > first_bin_idx; --idx) {
118  if (!is_almost_equal(accumulated_hist_white[idx], 0.0F)) {
119  last_bin_idx = idx;
120  break;
121  }
122  }
123 
124  float threshold = 0;
125  float max_ent = 0;
126  float ent_back;
127  float ent_obj;
128  float tot_ent;
129 
130  for (size_t idx = first_bin_idx; idx < last_bin_idx; ++idx) {
131 
132  /* Entropy of the background pixels */
133  ent_back = 0.0;
134 
135  for (size_t ih = 0; ih <= idx; ih++) {
136  if (!is_almost_equal(norm_hist[ih], 0.0F)) {
137  float c = norm_hist[ih] / accumulated_hist_black[idx];
138  ent_back -= c * std::log(c);
139  }
140  }
141 
142  ent_obj = 0.0;
143 
144  for (size_t ih = idx + 1; ih < num_bins; ih++) {
145  if (!is_almost_equal(norm_hist[ih], 0.0F)) {
146  float c = norm_hist[ih] / accumulated_hist_white[idx];
147  ent_obj -= c * std::log(c);
148  }
149  }
150 
151  /* Total entropy */
152  tot_ent = ent_back + ent_obj;
153 
154  if (max_ent < tot_ent) {
155  max_ent = tot_ent;
156  threshold = (float) idx;
157  }
158  }
159 
160  /**
161  * IMPORTANT: The histogram was "shrunk" to 256 values, i.e. float pixel value range was mapped to 256 brightness values.
162  * This "shrinking" step needs to be reverted so that the calculated threshold matches the original float image.
163  */
164  float th2 = min + (threshold / (float) num_bins) * (max - min);
165 
166  return th2;
167  }
