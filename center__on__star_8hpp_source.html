44 template<typename ImageType = float>
45 auto center_on_star(
46  const starmathpp::algorithm::Centroider<ImageType> &centroider) {
47  return ranges::views::transform(
48  [&](const cimg_library::CImg<ImageType> &&input_image) {
49 
50  auto opt_centroid = centroider.calculate_centroid(input_image);
51 
52  DEBUG_IMAGE_DISPLAY(input_image, "center_on_star_in",
53  STARMATHPP_PIPELINE_CENTER_ON_STAR_DEBUG);
54 
55  if (opt_centroid.has_value()) {
56  auto outer_roi = Rect<float>::from_center_point(
57  opt_centroid.value(), (float) input_image.width(),
58  (float) input_image.height()).template to<int>();
59 
60  // get_crop() by default applies a dirichlet boundary_condition (=0). There are other
61  // options as well. In this case, the desired behaviour is to assume that all pixel values
62  // where the defined sub-frame exceeds the image boundary are assumed to be 0.
63  //
64  // boundary_conditions = Can be { 0=dirichlet | 1=neumann | 2=periodic | 3=mirror }
65  //
66  // - Dirichlet means "0 outside image".
67  // - Neumann means "Nearest neighbor outside image" (i.e. null derivative)
68  // - Periodic means "Periodic"
69  // - Mirror means "Mirrored image outside".
70  //
71  // See https://github.com/GreycLab/CImg/issues/110
72  cimg_library::CImg<ImageType> centroid_sub_img(
73  input_image.get_crop(
74  outer_roi.x() /*x0*/, outer_roi.y() /*y0*/,
75  outer_roi.x() + outer_roi.width() - 1/*x1*/,
76  outer_roi.y() + outer_roi.height() - 1/*y1*/));
77 
78  DEBUG_IMAGE_DISPLAY(centroid_sub_img, "center_on_star_out",
79  STARMATHPP_PIPELINE_CENTER_ON_STAR_DEBUG);
80 
81  return centroid_sub_img;
82  } else {
83  throw starmathpp::algorithm::CentroiderException(
84  "Unable to determine centroid.");
85  }
86  }
87  );
88 }
89 }
